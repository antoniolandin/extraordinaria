% Theorems in TOC 
\makeatletter
\newcommand{\HelperDeclareTheorem}[4]{%
  \declaretheorem[
    numberlike=subsection,
    name=#3,
    refname={#1, #2},
    Refname={#3, #4},
    postheadhook={%
      \addcontentsline{toc}{subsection}{\protect\numberline{\thesubsection} #3 \ifdefempty{\thmt@optarg}{}{(\thmt@optarg)}}%
    }
  ]{%
    #1%
  }%
}
\makeatother
