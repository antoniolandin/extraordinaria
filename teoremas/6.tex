\begin{thm}
    Dado un grupo $G = \{X, \cdot\}$, para todos sus elementos $a\in G$ el inverso del inverso de $a$ es $a$.
    \begin{equation}
        \forall a \in G : (a^{-1})^{-1} = a 
    \end{equation}
\end{thm}

\begin{proof}
    \begin{equation}
        (a^{-1})^{-1} \cdot a^{-1} = e \iff  (a^{-1})^{-1} \cdot a^{-1} \cdot a = e \cdot a \iff (a^{-1})^{-1} = a
    \end{equation} 
\end{proof}
