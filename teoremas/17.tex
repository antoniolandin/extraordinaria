\begin{thm}
    Las clases laterales izquierdas $x_iH$ de un subgrupo $H$ normal, forman un grupo.
\end{thm}

\begin{proof}
    Las clases laterales izquierdas de un subgrupo $H$ cumplen con las tres propiedades de los grupos:

            \end{equation}
    \end{enumerate}
\end{proof}
