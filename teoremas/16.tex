\begin{thm}
    Para todo grupo $G = \{X, \cdot\}$ con un subgrupo $H \subset G$:
    \begin{equation}
        gHg^{-1} = H \quad g \in G \iff xH = Hx
    \end{equation}

    \begin{proof}
        $\implies$
        \begin{equation}
            gHg^{-1} = H \iff gHg^{-1}\cdot g = Hg \iff gH = Hg \quad h \in G   
        \end{equation}
    \end{proof}

    \begin{proof}
        $\impliedby$
        \begin{equation}
            xH = Hx \implies xH\cdot x^{-1} = Hx\cdot x^{-1} \iff xHx^{-1} = H \quad x \in G     
        \end{equation}
    \end{proof}
\end{thm}
