\begin{ce}
    $\mathbb{Z}[\sqrt{-7}]$ es D.F.U.
\end{ce}

\begin{proof}
    El elemento $8$ tiene dos factorizaciones:
    \begin{equation}
        \begin{split}
            & 8 = 2\cdot 2\cdot 2\\
            & 8 = (1 + \sqrt{-7})(1-\sqrt{-7})
        \end{split}
    \end{equation}

    Por lo tanto, hemos encontrado un elemento que no tiene una descomposición única.
\end{proof}
